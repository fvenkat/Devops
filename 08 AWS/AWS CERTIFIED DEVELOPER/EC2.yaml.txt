---
AWSTemplateFormatVersion: "2010-09-09"
Description: REDHAT EC2

Parameters:
  SSHKey:
    Description: Existing ssh key pair
    Type: AWS::EC2::KeyPair::KeyName

Resources:
  MyEc2Instance:
    Type: "AWS::EC2::Instance"
    Properties:
      InstanceType: t2.micro
      ImageId: ami-052c08d70def0ac62 # REDHAT LINUX Mumbai region
      KeyName: !Ref SSHKey
      SecurityGroups:
       - !Ref SSHSecurityGroup
      #Package installation
      UserData:
       Fn::Base64: |
	 #!/bin/bash -xe
	 yum update -y
	 yum install -y nginx
	 systemctl start nginx
	 systemctl enable nginx

  SSHSecurityGroup:
    Type: "AWS::EC2::SecurityGroup"
    Properties:
      GroupDescription: Enable SSH access via port 22
      SecurityGroupIngress:
       - CidrIp: 0.0.0.0/0
         FromPort: 22
         IpProtocol: tcp
         ToPort: 22

#EC2 Security Group
  ServerSecurityGroup:
    Type: "AWS::EC2::SecurityGroup"
    Properties:
      GroupDescription: Enable port 80
      SecurityGroupIngress:
      - IpProtocol: tcp
        FromPort: 80
        ToPort: 80
	CidrIp: 0.0.0.0/0

  #An Elastic IP for our EC2
  MyEIP:
    Type: "AWS::EC2::EIP"
    Properties:
      InstanceId:: !Ref MyEc2Instance
